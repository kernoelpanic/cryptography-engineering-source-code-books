\Input{A point $G$ on an elliptic curve and a scalar value $k$ (where $k=k_{t-1},...,k_1,k_0$ is the binary representation of $k$ starting with the most significant bit at $k_{t-1}$). }
\Output{The product $Q=kG$}

\BlankLine

\emph{First initialize $Q$ with the point at infinity as ''Null'' value}

$Q \leftarrow$ $\infty$

\For{$i \leftarrow 0$ \KwTo $t-1$} {
	\If{$k_i = 1$}{ 
		$Q \leftarrow Q + G$
	}
	$G \leftarrow 2G$
}
\Return $Q$

\BlankLine
